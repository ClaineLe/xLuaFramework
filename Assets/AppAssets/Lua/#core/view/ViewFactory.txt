local CSGameObject = CS.UnityEngine.GameObject
ViewFactory={
	
}


function ViewFactory:CreateView(viewName )
	local luaPresender = Framework.m_ViewMgr:LoadViewNode(viewName)
	return luaPresender
end

function ViewFactory:CreateViewAsync(viewName, callback )
	Framework.m_ViewMgr:LoadViewNodeAsync(viewName,function (presender)
		callback(presender)
	end)
end
