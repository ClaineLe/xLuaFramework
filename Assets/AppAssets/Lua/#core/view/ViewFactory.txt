local CSGameObject = CS.UnityEngine.GameObject
ViewFactory={
	
}


function ViewFactory:CreateView(viewName )
	local csPresender = Framework.m_ViewMgr:LoadViewNode(viewName)
	return csPresender.LuaPresender
end

function ViewFactory:CreateViewAsync(viewName, callback )
	Framework.m_ViewMgr:LoadViewNodeAsync(viewName,function (csPresender)
		callback(csPresender.LuaPresender)
	end)
end
