BaseView = {
	m_EventHandle = {},
}


function BaseView:Create( params  )
	local csView = params[0]
	local obj = {}
	setmetatable( obj, { __index = self })
	for k,v in pairs(csView._widgets) do
		obj[v.RefName] = v
	end

	for k,v in pairs(csView._subPresenders) do
		obj[v.m_View.RefName] = v.m_LuaTable
	end

	obj:init()
	return obj
end

end
